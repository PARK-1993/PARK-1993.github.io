JDK에서 가장 많이 쓰이는 패턴중 하나로 무언가 실시간으로 정보를 보낼 수 있고, 동적으로 실시간정보를 받을지 말지를 결정 할 수 있다. 한 객체의 상태가 바뀌면 그 객체에 의존하는 다른 객체들한테 연락이 가고 자동으로 내용이 갱신되는 일대다 의존성을 정의합니다. 여기서 한 객체를 subject라고 부르고 다른 여러객체들을 observer라고 한다.

구조

예

#include <iostream>
#include <string>
#include <vector>
typedef struct newProduct
{
	std::string productName;
	unsigned int price;
	float discountRate;
}NewProduct;

class CObserver {
	virtual void update(NewProduct&)= 0;
};
class CConcreateObserver : CObserver
{
public:
	void update(NewProduct& newProduct) override {
		showNewProductSpec(newProduct);
	}
private:
	void showNewProductSpec(NewProduct& newPD){
		std::cout << "<New Product>\nName = " << newPD.productName << std::endl;
		std::cout << "price = " << newPD.price << std::endl;
		std::cout << "discount rate = " << newPD.discountRate << std::endl;
	}

};


class CSubject {
public:
	virtual void registerObserver(CConcreateObserver*) = 0;
	virtual void removeObserver(CConcreateObserver*) = 0;
	virtual void notifyObserver() = 0;
};


class CConcreateSubject : CSubject {
public:
	void registerObserver(CConcreateObserver * o) override {
		lstObserver.push_back(o);
	}

	void removeObserver(CConcreateObserver*  o) override {
		auto iter = std::find(lstObserver.begin(), lstObserver.end(), o);
		if (iter != lstObserver.end()) {
			lstObserver.erase(iter);
		}
	}

	virtual void notifyObserver() override {
		for (auto iter = lstObserver.begin(); iter < lstObserver.end(); iter++)
		{
			(*iter)->update(m_NewProduct);
		}
	}
	void setNewProduct(const NewProduct& newProduct) {
		m_NewProduct = newProduct;
	}

private:
	std::vector<CConcreateObserver *> lstObserver;
	NewProduct m_NewProduct;


};

int main() {
	NewProduct nPD = { "CrazyBubbleBath", 500000, 0.5 };
	CConcreateObserver* objObserver = new CConcreateObserver;
	CConcreateSubject objSubject;
	objSubject.setNewProduct(nPD);
	objSubject.registerObserver(objObserver);
	objSubject.notifyObserver();

	return 0;
}
