Command Pattern이란

커맨드패턴을 이용하면 요구 사항을 객체로 캡슐화 할 수 있으며, 매개변수를 써서 여러가지 다른 요구 사항을 집어넣을 수도 있습니다. 또한 요청 내역을 큐에 저장하거나 로그로 기록할 수도 있으며, 작업취소 기능도 지원가능합니다. 헤드퍼스트에서 나온 정의이다. 어떠한 요구 사항, 명령을 각각 클래스로 만들어 캡슐화 하며, OCP를 준거한다. 요구나 명령을 프로그래밍할 경우가 있다면 CommandPattern을 사용하여 의존성은 낮추고 확장성을 높인 설계가 가능하다.

구조

커맨드 패턴에는 명령(command), 수신자(receiver), 발동자(invoker), 클라이언트(client)의 네개의 용어를 사용한다.

  • Command : 말그대로 명령이며, 인터페이스로서 execute라는 커맨드를 실행시키는 추상 메소드를 가지고 있다.
  • Invoker : 인보커란 호출자라고도 하며, Command를 가지고 있고, 여기서 execute 메소드를 호출하여 커맨드 객체에게 특정 작업을 수행해달라는 요구를 하게된다.
  • Receiver : 수신자이며, 요구 사항을 수행하기 위해 어떤 일을 처리해야 하는지 알고 있는 객체, 실제 요구의 대상이되는 객체이다. 예를들어 불을 켜다라는 명령에서는 전등이 요구, 명령의 대상인 리시버가 된다.
  • Client : 클라이언트는 Command를 생성하고, Receiver를 설정한다.

소스코드

#include<iostream>
#include<vector>
class Command
{
public:
	virtual void execute() = 0;
	virtual void undo() = 0;
};

class RemoteController
{
public:
	std::vector<Command*> ONcommand;
	std::vector<Command*> OFFcommand;
	Command* recentCommand;
	RemoteController()
	{
	}
	void setOnCommand(Command* command)
	{
		ONcommand.push_back(command);
	}

	void setOffCommand(Command* command)
	{
		OFFcommand.push_back(command);
	}
	void runON(unsigned short index)
	{
		ONcommand[index]->execute();
		recentCommand = ONcommand[index];
	}
	void runOFF(unsigned short index)
	{
		OFFcommand[index]->execute();
		recentCommand = OFFcommand[index];
	}

	void undo()
	{
		recentCommand->undo();
	}
};
class Light
{
public:
	void turnOn()
	{
		std::cout << "Turn On the light" << std::endl;
	}
	void turnOff()
	{
		std::cout << "Turn Off the light" << std::endl;
	}
};
class Radio
{
public:
	void turnOn()
	{
		std::cout << "Turn On the radio" << std::endl;
	}
	void turnOff()
	{
		std::cout << "Turn Off the radio" << std::endl;
	}
};


class LightOnCommand :public Command
{
private:
	Light* light;
public:
	LightOnCommand(Light* light)
	{
		this->light = light;
	}
	void execute() override
	{
		light->turnOn();
	}

	void undo() override
	{
		light->turnOff();
	}
};

class LightOffCommand :public Command
{
private:
	Light* light;
public:
	LightOffCommand(Light* light)
	{
		this->light = light;
	}
	void execute() override
	{
		light->turnOff();
	}
	void undo() override
	{
		light->turnOn();
	}
};

class RadioOnCommand :public Command
{
private:
	Radio* radio;
public:
	RadioOnCommand(Radio* radio)
	{
		this->radio = radio;
	}
	void execute() override
	{
		radio->turnOn();
	}
	void undo() override
	{
		radio->turnOff();
	}
};

class RadioOffCommand : public Command
{
private:
	Radio* radio;
public:
	RadioOffCommand(Radio* radio)
	{
		this->radio = radio;
	}
	void execute() override
	{
		radio->turnOff();
	}
	void undo() override
	{
		radio->turnOn();
	}
};


int main()
{
	// 리시버 객체생성
	Light* objLight = new Light;
	Radio* objRadio = new Radio;

	// 커맨드 객체에 리시버 설정
	LightOnCommand* objLightOnCommand = new LightOnCommand(objLight);
	LightOffCommand* objLightOffCommand = new LightOffCommand(objLight);
	RadioOnCommand* objRadioOnCommand = new RadioOnCommand(objRadio);
	RadioOffCommand* objRadioOffCommand = new RadioOffCommand(objRadio);

	// 인보커(리모컨) 객체 생성
	RemoteController* objRemoteController = new RemoteController();
	
	// 리모컨 버튼등록
	objRemoteController->setOnCommand(objLightOnCommand);
	objRemoteController->setOffCommand(objLightOffCommand);
	objRemoteController->setOnCommand(objRadioOnCommand);
	objRemoteController->setOffCommand(objRadioOffCommand);

	// 동작
	objRemoteController->runON(0);
	objRemoteController->runOFF(0);
	objRemoteController->runON(1);
	objRemoteController->runOFF(1);

	objRemoteController->undo();

	return 0;
}

위에 소스코드는 헤드퍼스트의 소스코드를 참고하였다.
