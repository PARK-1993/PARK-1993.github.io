팩토리패턴의 마지막 추상팩토리패턴이다. 먼저 정의를 보자

정의

추상팩토리패턴 에서는 인터페이스를 이용하여 서로 연관된,또는 의존하는 객체를 구상 클래스를 지정하지 않고도 생성할 수 있다. 정의만 보면 무슨말인지 잘 모르겠다. 일단 관련이있는 객체들을(제품군) 모아 생성,관리 하는 패턴이라고 생각하면 된다. 팩토리 메소드 패턴에서는 하나의 객체를 생성하기 위한 패턴이였지만(Pizza) 추상팩토리패턴은 연관이있는 여러 객체들을(제품군) 생성, 관리한다는 것이다.

구성

소스코드(팩토리메소드와 비교)

//Factory Method
class Client {
public:
    void anOperation() {
        Product p = factoryMethod();
        p.doSomething();
   }
protected:
    Product factoryMethod() {//or it can be abstract as well
        return new DefaultProduct();
   }
 }
// 팩토리패턴에서는 상속을 이용한다.
class NewClient : public Client {
protected:
    Product factoryMethod() {//overriding
        return new SpecificProduct();
   }
 }

 //Abstract Factory
class Client {
private:
// 추상팩토리 패턴에서는 구성을 사용한다.
    Factory factory;
public:
    Client (Factory factory) {
        this.factory = factory;
    } 
public:
    void anOperation() {
        ProductA p = factory.createProductA();
        p.doSomething();//other products and operations as well
    }
 }
class Factory {
   virtual ProductA createProductA() = 0;
   virtual ProductB createProductB() = 0;
 }
 //concrete factories also, implementing Factory interface
